2. MERGE SAP-FIRST (estrategia confirmada no RESEARCH):
   ```python
   merged = {}
   stats = {"sap_only": 0, "mercos_only": 0, "both_sap_base": 0}

   all_cnpjs = set(sap_data["cnpj_to_vendas"].keys()) | set(mercos_data["cnpj_to_vendas"].keys())
   for cnpj in all_cnpjs:
       sap = sap_data["cnpj_to_vendas"].get(cnpj)
       mercos = mercos_data["cnpj_to_vendas"].get(cnpj)

       if sap and mercos:
           # Ambos: SAP como base. Para meses onde SAP=0 mas Mercos>0, usar Mercos
           merged[cnpj] = [s if s > 0 else m for s, m in zip(sap, mercos)]
           stats["both_sap_base"] += 1
       elif sap:
           merged[cnpj] = sap
           stats["sap_only"] += 1
       else:
           merged[cnpj] = mercos
           stats["mercos_only"] += 1
   ```
